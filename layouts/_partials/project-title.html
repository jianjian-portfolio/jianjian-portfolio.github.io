{{- $title := .Title -}} {{- $main := $title -}} {{- $sub := "" -}} {{- if .Param "titleMultiline" -}} {{- if strings.Contains $title "：" -}} {{- $parts := split $title "：" -}} {{- $main = index $parts 0 -}} {{- $sub = strings.TrimSpace (delimit (after 1 $parts) "：") -}} {{- else if strings.Contains $title ": " -}} {{- $parts := split $title ": " -}} {{- $main = index $parts 0 -}} {{- $sub = strings.TrimSpace (delimit (after 1 $parts) ": ") -}} {{- end -}} {{- end -}} {{- if $sub -}} {{ $main }}{{ $sub }} {{- else -}} {{ $title }} {{- end -}}
